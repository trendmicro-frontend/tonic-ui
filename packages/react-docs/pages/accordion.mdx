# Accordion

## Import

```js
import {
  Accordion,
  AccordionItem,
  AccordionHeader,
  AccordionBody,
  AccordionToggle,
  AccordionCollapse,
} from '@tonic-ui/react';
```

## Usage

```jsx
<Accordion>
  <AccordionItem>
    <AccordionHeader>
      Header
    </AccordionHeader>
    <AccordionBody>
      Body
    </AccordionBody>
  </AccordionItem>
  <AccordionItem>
    <AccordionHeader>
      Header
    </AccordionHeader>
    <AccordionBody>
      Body
    </AccordionBody>
  </AccordionItem>
</Accordion>
```
