import { Box, Flex, Stack, Text } from "@tonic-ui/react";
import ThemeTokens from "@/components/ThemeTokens";
import { sizes as primitiveSizes } from "@tonic-ui/theme/src/v3/primitive/sizes";

export const SizesTable = () => {
  const entries = [
    { key: '0', rem: '0', px: '0px' },
    ...Object.entries(primitiveSizes.rem)
      .filter(([key]) => !key.endsWith('h'))
      .map(([key]) => ({ key, rem: primitiveSizes.rem[key], px: primitiveSizes.px[key] })),
  ];

  return (
    <table width="auto">
      <thead>
        <tr>
          <th style={{ textAlign: 'left' }}>Token</th>
          <th style={{ textAlign: 'right' }}>Relative length unit (rem)</th>
          <th style={{ textAlign: 'right' }}>Absolute length unit (px)</th>
        </tr>
      </thead>
      <tbody>
        {entries.map(({ key, rem, px }) => (
          <tr key={key}>
            <td style={{ textAlign: 'left' }}>
              <Flex columnGap="4x">
                <Text minWidth="10x">{key}</Text>
                <Box bg="teal:50" h="4x" w={key === '0' ? 0 : key} />
              </Flex>
            </td>
            <td style={{ textAlign: 'right' }}>{rem}</td>
            <td style={{ textAlign: 'right' }}>{px}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};

# sizes

By default, the values are proportional. Therefore, a `1x` spacing unit is equal to `0.25rem`, which translates to `4px` by default in common browsers.
