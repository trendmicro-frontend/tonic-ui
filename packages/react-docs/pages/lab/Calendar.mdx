# Calendar

## Import

```js
import {
  Calendar,
} from '@tonic-ui/react-lab';
```

## Usage

```jsx noInline
const FormGroup = (props) => (
  <Box mb="4x" {...props} />
);

const dateFormatOptions = [
  'yyyy-MM-dd', // ISO 8601
  'MM/dd/yyyy', // USA
  'dd.MM.yyyy', // Worldwide
  'yyyy/MM/dd', // East Asia
];

const firstDayOfWeekOptions = [
  0, // Sunday
  1, // Monday
  2, // Tuesday
  3, // Wednesday
  4, // Thursday
  5, // Friday
  6, // Saturday
];

const useSelection = (defaultValue) => {
  const [value, setValue] = React.useState(defaultValue);
  const changeBy = (value) => () => setValue(value);
  return [value, changeBy];
};

render(() => {
  const [value, setValue] = React.useState();
  const [dateFormat, changeDateFormatBy] = useSelection(dateFormatOptions[0]);
  const [firstDayOfWeek, changeWeekStartDayBy] = useSelection(0);

  return (
    <>
      <Box mb="4x">
        <Text fontSize="lg" lineHeight="lg">
          Selected date: {value ? dateFns.format(value, dateFormat) : null}
        </Text>
      </Box>
      <Calendar
        dateFormat={dateFormat}
        onChange={setValue}
        value={value}
        firstDayOfWeek={firstDayOfWeek}
      />
      <Divider my="4x" />
      <Box mb="4x">
        <Text fontSize="lg" lineHeight="lg">
          Calendar props
        </Text>
      </Box>
      <FormGroup>
        <Box mb="2x">
          <TextLabel>
            dateFormat
          </TextLabel>
        </Box>
        <ButtonGroup
          variant="secondary"
          css={{
            '> *:not(:first-of-type)': {
              marginLeft: -1
            }
          }}
        >
          {dateFormatOptions.map(value => (
            <Button
              key={value}
              selected={value === dateFormat}
              onClick={changeDateFormatBy(value)}
              minWidth="15x"
            >
              {value}
            </Button>
          ))}
        </ButtonGroup>
      </FormGroup>
      <FormGroup>
        <Box mb="2x">
          <TextLabel>
            firstDayOfWeek
          </TextLabel>
        </Box>
        <ButtonGroup
          variant="secondary"
          css={{
            '> *:not(:first-of-type)': {
              marginLeft: -1
            }
          }}
        >
          {[0, 1, 2, 3, 4, 5, 6].map(value => (
            <Button
              key={value}
              selected={value === firstDayOfWeek}
              onClick={changeWeekStartDayBy(value)}
              minWidth="15x"
            >
              {value}
            </Button>
          ))}
        </ButtonGroup>
      </FormGroup>
    </>
  );
});
```

## Props

### Calendar

| Name | Type | Default | Description |
| :--- | :--- | :------ | :---------- |
| dateFormat | string | 'yyyy-MM-dd' | The default date format is ISO 8601. <br/>'yyyy' = year<br/>'MM' = month<br/>'dd' = day |
| defaultValue | Date \| string | | The default selected date. If the `defaultValue` is a string, it will be parsed to a `Date` object in accordance with the `dateFormat`. |
| firstDayOfWeek | number | 0 | The first day of the week.<br/>0 = Sunday<br/>1 = Monday<br/>2 = Tuesday<br/>3 = Wednesday<br/>4 = Thursday<br/>5 = Friday<br/>6 = Saturday |
| onChange | (value) => void | | The callback function when the date is changed. The value is a `Date` object for the selected date. |
| value | Date \| string | | The selected date. If the `value` is a string, it will be parsed to a `Date` object in accordance with the `dateFormat`. |
