# Menu

A menu is used to display a list of options to the user. It appears when the user interacts with a button, link, or other control that triggers an action.

## Import

- `Menu`: The component that provides the menu functionality.
- `MenuButton`: A button that composes `Button`, `MenuToggle`, and `MenuToggleIcon` to create a menu button.
- `MenuContent`: An unstyled component that provides the menu content.
- `MenuDivider`: A divider that separates menu items.
- `MenuGroup`: A component that groups menu items.
- `MenuItem`: The individual menu items.
- `MenuList`: A styled `MenuContent` that wraps the menu items.
- `MenuToggle`: The toggle that opens the menu. This is usually a button or link.
- `MenuToggleIcon`: An indicator that appears next to the toggle. This is usually a chevron or arrow.
- `Submenu`: The component that provides the submenu functionality.
- `SubmenuList`: The list of menu items that appear in the submenu.
- `SubmenuToggle`: The toggle that opens the submenu when the menu item is hovered over.

```js
import {
  Menu,
  MenuButton,
  MenuContent,
  MenuDivider,
  MenuGroup,
  MenuItem,
  MenuList,
  MenuToggle,
  MenuToggleIcon,
  Submenu,
  SubmenuList,
  SubmenuToggle,
  useMenu,
  useSubmenu,
} from '@tonic-ui/react';
```

### Controlled menu

The menu opens over the anchor element by default, and closes when the user clicks outside of the menu.

```jsx noInline
render(() => {
  const [anchorEl, setAnchorEl] = React.useState(null);
  const [isOpen, setIsOpen] = React.useState(false);
  const [colorMode] = useColorMode();
  const [colorStyle] = useColorStyle({ colorMode });
  const [selectedValue, setSelectedValue] = React.useState(null);
  const handleOpenMenu = (event) => {
    setAnchorEl(event.currentTarget);
    setIsOpen(!isOpen);
  };
  const handleCloseMenu = () => {
    setAnchorEl(null);
    setIsOpen(false);
  };
  const handleClickMenuItem = (event) => {
    const value = event.target.getAttribute('value');
    setSelectedValue(value);
  };
  const handleMenuItemClick = (event) => {
    const value = event.target.getAttribute('value');
    console.log(`List item ${value} clicked`);
  };

  return (
    <Flex columnGap="4x" alignItems="center">
      <Button
        aria-controls="basic-menu"
        aria-haspopup="true"
        aria-expanded={isOpen ? 'true' : undefined}
        onClick={handleOpenMenu}
      >
        Options
      </Button>
      <Text>Selected: {selectedValue}</Text>
      {isOpen && (
        <Menu
          id="basic-menu"
          anchorEl={anchorEl}
          isOpen={isOpen}
          onClose={handleCloseMenu}
        >
          <MenuList
            onClick={handleClickMenuItem}
            width="max-content"
          >
            <MenuItem value={1}>
              List item 1
            </MenuItem>
            <MenuItem value={2}>
              List item 2
            </MenuItem>
            <MenuDivider />
            <Submenu>
              <SubmenuToggle>
                <MenuItem>
                  <Flex
                    alignItems="center"
                    columnGap="2x"
                    justifyContent="space-between"
                    width="100%"
                  >
                    Submenu
                    <Icon icon="angle-right" />
                  </Flex>
                </MenuItem>
              </SubmenuToggle>
              <SubmenuList width="max-content">
                <MenuItem value={3}>
                  List item 3
                </MenuItem>
                <MenuItem value={4}>
                  List item 4
                </MenuItem>
              </SubmenuList>
            </Submenu>
          </MenuList>
        </Menu>
      )}
    </Flex>
  );
});
```

### MenuButton

`MenuButton` composes `Button`, `MenuToggle`, and `MenuToggleIcon` to create a button that opens the menu. It's an useful alternative to `MenuToggle` when the toggle is a button.

```jsx disabled
const MenuButton = (props) => {
  const styleProps = {
    display: 'inline-flex',
    alignItems: 'center',
    justifyContent: 'space-between',
    columnGap: '1x',
  };

  return (
    <MenuToggle>
      {({ getMenuToggleProps }) => (
        <Button
          {...getMenuToggleProps()}
          {...styleProps}
          {...props}
        >
          {children && (
            <Box>{children}</Box>
          )}
          <MenuToggleIcon />
        </Button>
      )}
    </MenuToggle>
  );
};
```

### Placement

To position the menu relative to the toggle, pass a `placement` prop to `Menu`.

The value can be one of the following:
* `top`
* `top-start`
* `top-end`
* `bottom`
* `bottom-start`
* `bottom-end`

```jsx noInline
const useSelection = (defaultValue) => {
  const [value, setValue] = React.useState(defaultValue);
  const changeBy = (value) => () => setValue(value);
  return [value, changeBy];
};

const gridAreas = [
  ['top-start', 'top', 'top-end'],
  ['.', '.', '.'],
  ['.', 'center', '.'],
  ['.', '.', '.'],
  ['bottom-start', 'bottom', 'bottom-end'],
];

function Example() {
  const [placement, changePlacementBy] = useSelection('bottom-start');
  const gridTemplateAreas = gridAreas.map((row) => {
    const rowString = '"' + row.join(' ') + '"';
    return rowString;
  }).join(' ');

  return (
    <Flex>
      <Grid
        templateAreas={gridTemplateAreas}
        autoColumns="minmax(0, 1fr)"
        autoFlow="row"
        gap="2x"
      >
        {gridAreas.flat().map((value, key) => {
          if (value === '.') {
            return (<Box key={key} />);
          }

          if (value === 'center') {
            return (
              <Box key={key}>
                <Menu placement={placement}>
                  <MenuButton variant="secondary" width={150}>
                    <Text>Options</Text>
                  </MenuButton>
                  <MenuList>
                    <MenuItem>
                      List item 1
                    </MenuItem>
                    <MenuItem>
                      List item 2
                    </MenuItem>
                    <MenuItem>
                      List item 3
                    </MenuItem>
                  </MenuList>
                </Menu>
              </Box>
            );
          }

          const changePlacement = changePlacementBy(value);
          const onClick = () => {
            changePlacement();
          };

          return (
            <Box key={key}>
              <Button
                selected={value === placement}
                onClick={onClick}
                width="100%"
              >
                {value}
              </Button>
            </Box>
          );
        })}
      </Grid>
    </Flex>
  );
}

render(<Example />);
```

### Change menu position

The `offset` prop can be used to change the menu position. It accepts an array with two numbers in the form of `[skidding, distance]`.

**Skidding**

The first number, `skidding`, displaces the menu along the toggle. This value is specified in pixels.

**Distance**

The second number, `distance`, controls the distance between the menu and the toggle. This value is specified in pixels.

```jsx
<Menu offset={[16, 8]}>
  <MenuButton variant="secondary">
    <Text>Options</Text>
  </MenuButton>
  <MenuList>
    <MenuItem>
      List item 1
    </MenuItem>
    <MenuItem>
      List item 2
    </MenuItem>
    <MenuItem>
      List item 3
    </MenuItem>
  </MenuList>
</Menu>
```

### Scrolling

To make the menu scrollable, pass `overflow="auto"` and `maxHeight` to `MenuList`. The menu will become scrollable when the menu items exceed the height of the menu.

```jsx
<Menu display="block">
  <MenuButton variant="secondary" width={200}>
    <Text>Options</Text>
  </MenuButton>
  <MenuList
    maxHeight={200}
    overflow="auto"
    width={200}
  >
    {Array.from({ length: 100 }).map((_, key) => (
      <MenuItem key={key}>
        List item {key + 1}
      </MenuItem>
    ))}
  </MenuList>
</Menu>
```

You can also use the `Scrollbar` component to override the default scrollbar.

```jsx
<Menu display="block">
  <MenuButton variant="secondary" width={200}>
    <Text>Options</Text>
  </MenuButton>
  <MenuList width={200}>
    <Scrollbar height={200} overflowY="visible">
      {Array.from({ length: 100 }).map((_, key) => (
        <MenuItem key={key}>
          List item {key + 1}
        </MenuItem>
      ))}
    </Scrollbar>
  </MenuList>
</Menu>
```

### Input controls

Set `closeOnSelect` to `false` to keep the menu open after selecting a menu item.

You can use arrow keys to navigate the menu, and press `Enter` or `Space` to select an item.

```jsx
function menuWithCheckbox() {
  const [checkedItems, setCheckedItems] = React.useState(['host', 'ip']);
  const handleCheckboxGroupChange = (value) => {
    setCheckedItems(value);
  };
  const handleKeyDown = (event) => {
    if (event.key === 'Enter' || event.key === ' ') {
      const el = event.currentTarget.querySelector('input');
      if (el) {
        const value = el.value;
        const items = [...checkedItems];
        items.indexOf(value) === -1 ? items.push(value) : items.splice(items.indexOf(value), 1);
        setCheckedItems(items);
      }
    }
  };

  return (
    <Menu closeOnSelect={false}>
      <MenuButton variant="secondary">
        <Text>Selected ({checkedItems.length})</Text>
      </MenuButton>
      <MenuList width="max-content">
        <CheckboxGroup
          size="sm"
          value={checkedItems}
          onChange={handleCheckboxGroupChange}
        >
          <MenuItem onKeyDown={handleKeyDown}>
            <Checkbox value="host">
              Endpoint name</Checkbox>
          </MenuItem>
          <MenuItem onKeyDown={handleKeyDown}>
            <Checkbox value="ip">
              IP address
            </Checkbox>
          </MenuItem>
          <MenuItem onKeyDown={handleKeyDown}>
            <Checkbox value="os">
              Operating system
            </Checkbox>
          </MenuItem>
        </CheckboxGroup>
      </MenuList>
    </Menu>
  );
}
```

```jsx noInline
render(() => {
  const textRef = React.useRef();
  const [isTextOverflow, setIsTextOverflow] = React.useState(false);
  const searchInputRef = React.useRef();
  const [checkedItems, setCheckedItems] = React.useState([]);

  const handleCheckboxGroupChange = React.useCallback((value) => {
    setCheckedItems(value);
  }, []);

  React.useLayoutEffect(() => {
    const el = textRef.current;
    if (el) {
      // overflow appears if the scrollWidth is greater than the clientWidth
      const willTextOverflow = el.clientWidth < el.scrollWidth;
      if (willTextOverflow !== isTextOverflow) {
        setIsTextOverflow(willTextOverflow);
      }
    }
  });

  const checkedTextItems = checkedItems.map(x => countryObject[x]);

  return (
    <Menu
      closeOnSelect={false}
    >
      <MenuButton
        variant="secondary"
      >
        <Flex alignItems="center" columnGap="2x">
          <Tooltip label={checkedTextItems.join(', ')}>
            <>
              <Truncate
                ref={textRef}
                textAlign="left"
                width={260}
              >
                {checkedTextItems.join(', ')}
              </Truncate>
              {isTextOverflow && (
                `(${checkedTextItems.length})`
              )}
            </>
          </Tooltip>
        </Flex>
      </MenuButton>
      <MenuList>
        <SearchInput
          ref={searchInputRef}
          placeholder="Search"
          mb="2x"
          mx="3x"
        />
        <Link
          mx="3x"
          mb="2x"
        >
          Select all
        </Link>
        <CheckboxGroup
          size="sm"
          value={checkedItems}
          onChange={handleCheckboxGroupChange}
        >
          <Box
            maxHeight={400}
            overflowY="auto"
          >
            {items.map(item => {
              return (
                <MenuItem
                  key={item.value}
                >
                  <Checkbox
                    value={item.value}
                    width="100%"
                  >
                    {item.label}
                  </Checkbox>
                </MenuItem>
              );
            })}
          </Box>
        </CheckboxGroup>
      </MenuList>
    </Menu>
  );
});
```
