# useOnce

A custom Hook that runs a callback at most once.

## Import

```js
import { useOnce } from '@tonic-ui/react-hooks';
```

### useOnce

```js
import useOnceWhen from './useOnceWhen';

const useOnce = (callback) => {
  useOnceWhen(callback, true);
};
```

## Usage

```jsx
function Component() {
  const [value, setValue] = React.useState(0);
  const ref = React.useRef(0);

  useOnce(() => {
    console.log('This will run only once before the initial render');
    ref.current++;
  });

  return (
    <>
      <Box mb="4x">
        Callback called: {ref.current}
      </Box>
      <Button onClick={() => setValue(value => value + 1)}>
        Click Me
      </Button>
    </>
  );
}
```
