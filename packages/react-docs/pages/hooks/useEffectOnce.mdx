# useEffectOnce

A modified version of the `useEffect` Hook that runs an effect only once.

## Import

```js
import { useEffectOnce } from '@tonic-ui/react-hooks';
```

### useEffectOnce

```js
import { useEffect } from 'react';

const useEffectOnce = (effect) => {
  // eslint-disable-next-line react-hooks/exhaustive-deps
  useEffect(effect, []);
};
```

## Usage

```jsx
function Component() {
  const [value, setValue] = React.useState(0);

  React.useEffect(() => {
    console.log('useEffect is triggered when value changes', { value });
  }, [value]);

  useEffectOnce(() => {
    console.log('useEffectOnce is triggered only once', { value });
  });

  return (
    <>
      <Box mb="2x">
        {value}
      </Box>
      <Button onClick={() => setValue(value => value + 1)}>
        Click Me
      </Button>
    </>
  );
}
```
