# useEventListener

A custom Hook to manage browser event listeners.


```js
// import
import { useEventListener } from '@tonic-ui/react-hooks';

// usage
useEventListener(target, eventName, eventHandler, [options]);
```

### Parameters

| Name | Type | Default | Description |
| :--- | :--- | :------ | :---------- |
| target | HTMLElement | | The target element to attach the event listener to. |
| eventName | string | | The name of the event to listen for. |
| eventHandler | `(event: MouseEvent) => void` | | The event handler to call when the event is fired. |
| options? | object | | Optional options to pass to the event handler. |

### Example

```jsx
function Component() {
  const buttonRef = React.useRef(null);

  const onScroll = (event) => {
    console.log('document scrolled', event);
  };

  const onClick = (event) => {
    console.log('button clicked', event);
  };

  // example with document based event
  useEventListener(
    () => (buttonRef.current.ownerDocument || document),
    'scroll',
    onScroll,
  );

  // example with element based event
  useEventListener(
    () => buttonRef.current,
    'click',
    onClick,
  );

  return (
    <Button ref={buttonRef}>
      Click Me
    </Button>
  );
}
```
