# Shadows

## Design Tokens

<ThemeParser theme="shadows" />

> Shadow colors are defined in the color style object, not in the theme. Use the `useColorStyle` Hook to get the shadow color.

## Color Style

```js
const [colorMode] = useColorMode();
const [colorStyle] = useColorStyle({ colorMode });

console.log(colorStyle.shadow.thin);
console.log(colorStyle.shadow.medium);
console.log(colorStyle.shadow.thick);
```

### Dark Mode

```js
shadow: {
  thin: '0 2px 8px 0 rgba(0, 0, 0, 0.48), 0 1px 2px 0 rgba(0, 0, 0, 0.16)',
  medium: '0 4px 16px 0 rgba(0, 0, 0, 0.48), 0 2px 4px 0 rgba(0, 0, 0, 0.16)',
  thick: '0 8px 32px 0 rgba(0, 0, 0, 0.48), 0 4px 8px 0 rgba(0, 0, 0, 0.16)',
}
```

### Light Mode

```js
shadow: {
  thin: '0 2px 8px 0 rgba(0, 0, 0, 0.16), 0 1px 2px 0 rgba(0, 0, 0, 0.08)',
  medium: '0 4px 16px 0 rgba(0, 0, 0, 0.16), 0 2px 4px 0 rgba(0, 0, 0, 0.08)',
  thick: '0 8px 32px 0 rgba(0, 0, 0, 0.16), 0 4px 8px 0 rgba(0, 0, 0, 0.08)',
}
```

## Examples

```jsx noInline
function Example() {
  const [colorMode] = useColorMode();
  const [colorStyle] = useColorStyle({ colorMode });
  const backgroundColor = colorStyle.background.secondary;
  const color = colorStyle.color.secondary;
  const capitalizeFirstLetter = str => str.charAt(0).toUpperCase() + str.slice(1);

  return (
    <Flex flexDirection="row" columnGap="6x">
      {['thin', 'medium', 'thick'].map(key => (
        <Flex
          key={key}
          width={256}
          height={128}
          backgroundColor={backgroundColor}
          color={color}
          boxShadow={colorStyle.shadow[key]}
          alignItems="center"
          justifyContent="center"
        >
          <Text fontSize="md" lineHeight="md">
            {capitalizeFirstLetter(key)}
          </Text>
        </Flex>
      ))}
    </Flex>
  );
}

render(<Example />);
```
