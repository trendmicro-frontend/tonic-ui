import { Box, Stack } from "@trendmicro/react-styled-ui";
import ThemeParser from "../components/ThemeParser";

# Borders

* `borders`
* `borderStyles`
* `borderWidths`
* `radii` (`border-radius`)

## Border

Defaults to an empty object.

<ThemeParser theme="borders" />

## Border Style

Defaults to an empty object.

<ThemeParser theme="borderStyles" />

## Border Width

<ThemeParser theme="borderWidths" />

```jsx
<Stack direction="column" spacing="8x" alignItems="center">
  <Stack direction="row" spacing="6x">
    <Box border={1} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderTop={1} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderRight={1} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderBottom={1} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderLeft={1} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
  </Stack>
  <Stack direction="row" spacing="6x">
    <Box border={2} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderTop={2} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderRight={2} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderBottom={2} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
    <Box borderLeft={2} borderColor="gray:40" width="32x" height="12x" bg="gray:80"/>
  </Stack>
</Stack>
```

## Border Radius

<ThemeParser theme="radii" />

```jsx
<Stack direction="row" spacing="6x" justifyContent="center">
  <Box border={1} borderColor="gray:40" borderRadius="circle" width="18x" height="18x" bg="gray:80"/>
  <Box border={1} borderColor="gray:40" borderRadius="sm" width="18x" height="18x" bg="gray:80"/>
  <Box border={1} borderColor="gray:40" borderRadius="md" width="18x" height="18x" bg="gray:80"/>
  <Box border={1} borderColor="gray:40" borderRadius="lg" width="18x" height="18x" bg="gray:80"/>
</Stack>
```
